## 3. 后处理策略优化

### 3.1 任务特定的后处理

在 `src/openpi/policies/policy.py` 的 `Policy.post_process` 方法中添加：

```python
def post_process(self, obs: dict, outputs: dict) -> dict:
    task_name = obs.get("task_name", None)

    if task_name == "sorting_packages":
        # 针对排序任务优化腰部动作
        outputs = self._optimize_waist_actions(outputs)
    elif task_name == "pouring_workpiece":
        # 针对倒水任务优化手臂动作
        outputs = self._optimize_arm_orientation(outputs)
    elif task_name == "opening_door":
        # 针对开门任务优化抓取力
        outputs = self._optimize_grip_strength(outputs)

    return outputs

def _optimize_waist_actions(self, outputs):
    """优化腰部动作以更平滑地转动"""
    waist_actions = outputs["actions"][:, 16:20]
    # 应用平滑滤波
    for dim in range(4):
        waist_actions[:, dim] = self._exponential_smoothing(waist_actions[:, dim])
    outputs["actions"][:, 16:20] = waist_actions
    return outputs

def _exponential_smoothing(self, actions, alpha=0.8):
    """指数平滑"""
    smoothed = np.zeros_like(actions)
    smoothed[0] = actions[0]
    for i in range(1, len(actions)):
        smoothed[i] = alpha * actions[i] + (1 - alpha) * smoothed[i-1]
    return smoothed

def _optimize_arm_orientation(self, outputs):
    """优化手臂方向以适应倒水任务"""
    arm_actions = outputs["actions"][:, :14]
    # 针对特定维度添加微调
    arm_actions[:, 5:8] *= 0.9  # 降低腕部动作幅度
    outputs["actions"][:, :14] = arm_actions
    return outputs

def _optimize_grip_strength(self, outputs):
    """优化抓取力"""
    grip_dim = 13  # 根据实际动作维度调整
    outputs["actions"][:, grip_dim] = np.maximum(
        outputs["actions"][:, grip_dim], 0.5  # 保证最小抓取力
    )
    return outputs
```

### 3.2 动作平滑和过滤

```python
def _smooth_actions(self, actions):
    """简单的滑动平均平滑"""
    return np.convolve(actions, np.ones(3)/3, mode='same')

def _median_filter(self, actions, window_size=5):
    """中值滤波，减少异常值"""
    result = np.copy(actions)
    half_window = window_size // 2
    for i in range(half_window, len(actions) - half_window):
        result[i] = np.median(actions[i-half_window:i+half_window+1])
    return result
```

```python
# 核心思路：冻结预训练模型，训练一个轻量级策略网络

from rlinf import Policy as RLPolicy
import dataclasses

class ACoTWithRL(Policy):
    def __init__(self, acot_model, rl_policy=None):
        self.acot = acot_model
        self.rl = rl_policy  # RL 修正网络

    def infer(self, obs):
        base_actions = self.acot.infer(obs)

        if self.rl is not None:
            # 使用 RL 网络修正动作
            corrected_actions = self.rl.infer(obs, base_actions)
            return corrected_actions

        return base_actions

@dataclasses.dataclass
class ActionCorrectionNetwork(nnx.Module):
    """轻量级动作修正网络"""
    state_dim: int
    action_dim: int
    hidden_dim: int = 256

    def __init__(self, rngs: nnx.Rngs):
        self.fc1 = nnx.Linear(self.state_dim + self.action_dim, self.hidden_dim, rngs=rngs)
        self.fc2 = nnx.Linear(self.hidden_dim, self.hidden_dim, rngs=rngs)
        self.fc3 = nnx.Linear(self.hidden_dim, self.action_dim, rngs=rngs)

    def __call__(self, state, base_action):
        x = jnp.concatenate([state, base_action], axis=-1)
        x = nnx.swish(self.fc1(x))
        x = nnx.swish(self.fc2(x))
        delta = self.fc3(x)  # 预测动作增量
        return base_action + 0.1 * delta  # 小幅度修正
```

```python
def online_train_step(self, obs, actions, reward, rng):
    """在线训练步骤"""
    # 1. 训练奖励预测器
    reward_pred = self._reward_predictor(obs)
    reward_loss = self._compute_reward_loss(reward_pred, reward)

    # 2. 更新策略
    policy_loss = self._compute_policy_loss(obs, actions, reward_pred)

    # 3. 组合损失并更新
    total_loss = reward_loss + policy_loss
    return total_loss

def adaptive_policy(self, obs):
```

## 5. 集成学习方法

### 5.1 模型集成

```python
class EnsemblePolicy(Policy):
    """多模型集成策略"""

    def __init__(self, models: list[Policy], weights: list[float] = None):
        self.models = models
        self.weights = weights if weights is not None else [1.0 / len(models)] * len(models)

    def infer(self, obs):
        predictions = []

        # 收集所有模型的预测
        for model in self.models:
            pred = model.infer(obs)
            predictions.append(pred["actions"])

        # 加权平均
        final_actions = np.zeros_like(predictions[0])
        for weight, pred in zip(self.weights, predictions):
            final_actions += weight * pred

        return {"actions": final_actions}

def ensemble_infer(self, obs):
    """集成推理 - 投票或平均"""
    predictions = []
    for model in self.ensemble_models:
        predictions.append(model.infer(obs))

    # 集成策略（投票或平均）
    final_actions = np.mean(predictions, axis=0)
    return final_actions
```

### 5.2 输出空间集成

```python
def acot_ensemble(self, obs):
    """ACoT 特定的集成策略"""
    # 粗动作集成
    coarse_preds = [m._sample_coarse_actions(obs) for m in self.ensemble]
    coarse_final = self._voting_strategy(coarse_preds)

    # 细动作集成（基于粗动作）
    fine_preds = [m._sample_fine_actions(obs, coarse_final) for m in self.ensemble]
    fine_final = self._mean_strategy(fine_preds)

    return fine_final

def _voting_strategy(self, predictions):
    """基于置信度的投票策略"""
    confidences = self._estimate_confidence(predictions)
    best_idx = np.argmax(confidences)
    return predictions[best_idx]

def _mean_strategy(self, predictions):
    """简单平均策略"""
    return np.mean(predictions, axis=0)

def _estimate_confidence(self, predictions):
    """估计每个预测的置信度"""
    # 使用预测之间的方差作为不确定性的度量
    mean_pred = np.mean(predictions, axis=0)
    variances = np.mean([np.sum((p - mean_pred) ** 2) for p in predictions], axis=0)
    confidences = 1.0 / (variances + 1e-6)  # 低方差 = 高置信度
    return confidences
```

## 6. 推理优化

### 6.1 置信度阈值

```python
def infer_with_confidence(self, obs):
    """带置信度估计的推理"""
    outputs = self.infer(obs)

    # 计算动作置信度
    conf = self._compute_confidence(obs, outputs)

    if conf < 0.3:
        # 低置信度时使用更保守的策略
        outputs = self._fallback_policy(obs)

    return outputs

def _compute_confidence(self, obs, outputs):
    """计算置信度分数"""
    # 方法 1: 基于动作的方差
    action_variance = np.var(outputs["actions"], axis=0)
    confidence = 1.0 / (np.mean(action_variance) + 1e-6)

    # 方法 2: 基于图像特征的一致性
    # img_features = self._extract_image_features(obs)
    # confidence = self._confidence_model(img_features)

    return confidence

def _fallback_policy(self, obs):
    """回退策略 - 更保守的动作"""
    # 可以是简化的动作、停止动作、或重复上一个安全动作
    safe_action = np.zeros((self.action_horizon, self.action_dim))
    return {"actions": safe_action}
```

### 6.2 动态 Horizon 调整

```python
def dynamic_horizon_infer(self, obs):
    """根据任务复杂度动态调整动作 horizon"""
    task_complexity = self._predict_task_complexity(obs)

    if task_complexity > 0.7:
        # 更复杂的任务使用更长的 horizon
        config = acot_vla.ACOTConfig(
            coarse_action_horizon=60,
            action_horizon=30
        )
    elif task_complexity > 0.3:
        # 中等复杂度
        config = acot_vla.ACOTConfig(
            coarse_action_horizon=40,
            action_horizon=20
        )
    else:
        # 简单任务
        config = acot_vla.ACOTConfig(
            coarse_action_horizon=30,
            action_horizon=15
        )

    return self._infer_with_config(obs, config)

def _predict_task_complexity(self, obs):
    """预测任务复杂度（0-1）"""
    prompt = obs.get("prompt", "")

    # 简单的启发式规则
    complexity = 0.5  # 默认中等复杂度

    # 基于任务关键词
    if "sort" in prompt.lower() or "multiple" in prompt.lower():
        complexity = 0.8  # 复杂任务
    elif "pour" in prompt.lower() or "scoop" in prompt.lower():
        complexity = 0.3  # 简单任务

    return complexity
```

```python
class HybridPolicy(Policy):
    """混合策略 - 简单任务用 Pi0/0.5，复杂任务用 ACoT"""

    def __init__(self, pi05_policy, acot_policy):
        self.pi05_policy = pi05_policy
        self.acot_policy = acot_policy
        self.simple_tasks = [
            "pouring_workpiece",
            "scooping_popcorn",
            "turning_doorknob"
        ]

    def infer(self, obs):
        task_name = obs.get("task_name", "")

        if any(simple in task_name for simple in self.simple_tasks):
            # 简单任务使用 Pi0.5（更快）
            return self.pi05_policy.infer(obs)
        else:
            # 复杂任务使用 ACoT（更准确）
            return self.acot_policy.infer(obs)
```

## 实施优先级建议

### 优先级排序（从易到难）

1. **数据增强**（最易实现，收益高）
   - 增加 prompt 多样性
   - 优化 action shift 参数
   - 添加图像增强

2. **超参数优化**（中等难度，收益稳定）
   - 调整学习率
   - 优化 batch size
   - 调整训练步数

3. **后处理策略**（易实现，针对特定任务收益高）
   - 任务特定的动作优化
   - 动作平滑和过滤
   - 状态相关的动作修正

4. **隐式提取器替换**（中等难度，架构改进）
   - 尝试不同的 IAR 提取器
   - 调整动作专家大小
   - 配置不同的 horizon

5. **RL 微调**（较难，需要 RL 知识，但收益高）
   - 实现轻量级修正网络
   - 添加在线策略更新
   - 集成 RL 奖励机制

6. **集成学习**（中等难度，需要多模型资源）
   - 多模型集成
   - 输出空间集成
   - 动态模型选择

7. **架构重设计**（难，需要深入理解）
   - 修改推理机制
   - 重新设计 EAR/IAR 交互
   - 探索新的融合策略

### 具体实施步骤

**第一阶段：基础优化**
- [ ] 优化数据和超参数（基础）
- [ ] 针对特定任务优化策略（如 waist 动作、pouring）

**第二阶段：架构优化**
- [ ] 尝试不同的隐式提取器
- [ ] 对比 EAR/IAR 不同组合
- [ ] 调整 LoRA 配置

**第三阶段：高级策略**
- [ ] 考虑集成和混合策略（提升稳定性）
